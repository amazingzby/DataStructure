#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;

int main() {
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */
    int N,Q;
    cin>>N>>Q;
    vector<vector<int> >seqList(N);
    int lastAnts=0;
    for(int i=0;i<Q;++i)
    {
        int query;
        long long x,y;
        cin>>query>>x>>y;
        int m=(x^lastAnts)%N;
        int n=int(y%seqList[m].size());
        switch(query)
        {
          case 1:
          {
              seqList[m].push_back(y);
              break;
          }
          case 2:
          {
              lastAnts=seqList[m][n];
              cout<<lastAnts<<endl;
              break;
          }
        }
    }
    
    return 0;
}
